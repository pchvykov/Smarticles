 1653   static void pivotArm(real_T al, real_T th, real_T c_A, real_T out[3])
 1654   {
 1655     real_T secth;

 1657     real_T py;
 1658     real_T pylnpy2;
 1659   
 1660     /* response when force applied to the right arm or corner */
 1661     /* in: arm angle, force angle, arm length (in units of B) */
 1662     /* out: [px,py,f] = pivot coordinates (in units of B) and critical force */
 1663     /* required to move (in units of friction force; f>0 => CCW rotation) */
 1664     secth = 1.0 / muDoubleScalarCos(th);
 1665     tanth = muDoubleScalarTan(th);

 1667     /* the constant out front quantifies how far off the pivot can be (larger=>further away). "right" value: 0.081 */

 1673     /* differentiable approximation for py*log(py^2) for py in (-0.5,0.5) */

 1675     out[1] = py;
 1676     out[2] = -secth * pylnpy2;
